create table users(id serial primary key,name text not null,email text,created_at timestamp default current_timestamp);
create table orders(id serial primary key,user_id int,amount numeric(10,2),status text,created_at timestamp default current_timestamp,foreign key(user_id) references users(id) on delete cascade on update cascade);
